# We set:
# - TBB_INCLUDE_DIR
# - TBB_LIBRARIES
# - TBB_FOUND
# Taking into account:
# - TBB_PREFIX

IF(WIN32)
  FIND_PATH(TBB_INCLUDE_DIR
      tbb
    PATHS
      ${TBB_PREFIX}/include
      C:/local/tbb44_20150728oss/include
  )
ELSE(WIN32)
  FIND_PATH(TBB_INCLUDE_DIR
      tbb
    PATHS
      ${TBB_PREFIX}/include
      /usr/include
      /usr/local/include
      /opt/include
      /opt/local/include
      ~/tbb44_20150728oss/include
)
ENDIF(WIN32)

IF(WIN32)
  FIND_LIBRARY(TBB_LIBRARIES
    NAMES
      tbb
    PATHS
      ${TBB_PREFIX}/lib
      c:/local/tbb44_20150728oss/lib/intel64/vc12
  )
ELSE(WIN32)
  FIND_LIBRARY(TBB_LIBRARIES
    NAMES
      tbb
    PATHS
      ${TBB_PREFIX}/lib
      ${USERLIB_PREFIX}/lib
      /opt/lib
      /opt/local/lib
      ~/tbb44_20150728oss/lib
  )
ENDIF(WIN32)

IF(TBB_INCLUDE_DIR
  AND TBB_LIBRARIES)
    SET(TBB_FOUND TRUE)
    MESSAGE(STATUS "Found Intel Threading Building Blocks (Intel TBB) library")
ELSE(TBB_LIBRARIES)
    SET(TBB_FOUND FALSE)
    MESSAGE(STATUS "Intel TBB library not found - set TBB_PREFIX")
ENDIF(TBB_INCLUDE_DIR
  AND TBB_LIBRARIES)
